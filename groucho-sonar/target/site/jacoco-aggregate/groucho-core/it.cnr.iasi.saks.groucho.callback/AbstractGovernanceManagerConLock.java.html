package it.cnr.iasi.saks.groucho.callback;

import java.util.Set;
import java.util.concurrent.locks.Condition;
import java.util.concurrent.locks.Lock;
import java.util.concurrent.locks.ReentrantLock;

import it.cnr.iasi.saks.groucho.common.Context;
import it.cnr.iasi.saks.groucho.isolation.RuntimeEnvironmentShield;

public abstract class AbstractGovernanceManagerConLock implements ThreadHarness {
	
	private volatile boolean isInVivoTestingEnabled;

	private RuntimeEnvironmentShield environmentShield;

	private static volatile Object INTERNAL_LOCK = new Object();
	private static volatile Object THREAD_LOCKER = new Object();
			
	private static volatile ReentrantLock LOCK = new ReentrantLock();
	private static volatile Condition OTHER_ACTIVE_THREADS  = LOCK.newCondition(); 
	private static volatile Condition notEmpty = LOCK.newCondition();
	   
	private static volatile int THREAD_COUNTER = 1; // If this variable exists, it means that there is at least 1 thread. So it must start from 1
	
	public AbstractGovernanceManagerConLock(){
		this.isInVivoTestingEnabled = false;
		this.environmentShield = new RuntimeEnvironmentShield();		
	} 
	
	public synchronized void runInvivoTestingSession(Context context){
		
		if (this.checkActivation(context)){
			this.isInVivoTestingEnabled = true;
//			this.waitOtherThreadsPaused();
			
			System.out.println("the checkpoint of the considered object should be applied here ... ");		
			this.environmentShield.applyCheckpointOnContext(context);
			
			this.doInvivoTestingSession(context);
			
			System.out.println("... while its rollback is here.");		
			this.environmentShield.applyRollbackOnContext(context);
			
			this.notifyOtherThreads();
		}
		this.isInVivoTestingEnabled = false;
	}
	
	public void runSimplifiedInvivoTestingSession(int ID){
		boolean gotTheLock = LOCK.tryLock();
		if (gotTheLock){
			try{
				if (this.checkActivation(null)){
					this.isInVivoTestingEnabled = true;
					this.waitOtherThreadsPaused(ID);
			
					System.out.println("the checkpoint of the considered object should be applied here ... ");		
			
					System.out.println("... while its rollback is here.");		
			
					this.notifyOtherThreads();
				}
			}finally{	
				this.isInVivoTestingEnabled = false;
				LOCK.unlock();
			}
		}	
	}

	private boolean checkActivation(Context context){
		boolean exitStatus;
		
		// TO-DO this has to be improved, so that to check if there exists any precondition to the activation of an Invivo Testing Session
		exitStatus = true;
				
		exitStatus = exitStatus && this.evaluateActivation(context);		
		return exitStatus;
	}

		try {
//			synchronized (INTERNAL_LOCK) {
			int count = 1; // the current thread should not be considered, so counter starts from 1 and condition is >=
			while ( ! (count >= THREAD_COUNTER) ){
System.out.println("["+ID+"] Count: " + count + ", Thread: "+ THREAD_COUNTER );					
//					INTERNAL_LOCK.wait();
					OTHER_ACTIVE_THREADS.await();
					count ++;
				}
//			}	
		} catch (InterruptedException e) {
			// TODO Auto-generated catch block
			e.printStackTrace();
		}
	}
	
	private void notifyOtherThreads(){
//		Set<Thread> threadSet = Thread.getAllStackTraces().keySet();
//System.out.println("Notifico tutti");
//		for (Thread thread : threadSet) {
//System.out.println("Notifico ... ");
//			thread.notify();

		synchronized (THREAD_LOCKER) {
			System.out.println("Notifico tutti");
			THREAD_LOCKER.notifyAll();
		}
	}

	private void signalFromAnActiveThreads(){
		LOCK.lock();
		try{
			OTHER_ACTIVE_THREADS.signal();
		} finally{
			LOCK.unlock();
		}
	}

		if (this.isInVivoTestingEnabled){
			System.out.println("Ci Provo ... ");

	}

	public synchronized void incThreads(){
		THREAD_COUNTER++;
	}

	public void decThreads(){
		THREAD_COUNTER--;
		this.signalFromAnActiveThreads();

	protected abstract boolean evaluateActivation (Context context);
